let balance = 100; document.getElementById("flipButton").addEventListener("click", function () { const betAmount = parseInt(document.getElementById("betAmount").value); const betChoice = document.getElementById("betChoice").value; // Check if the bet amount is valid and the user has enough balance if (betAmount <= 0 || betAmount > balance) { alert("Invalid bet amount!"); return; } // Flip the coin (randomly pick 'heads' or 'tails') const coinResult = Math.random() < 0.5 ? 'heads' : 'tails'; // Update game result let resultText = ''; if (coinResult === betChoice) { balance += betAmount; // User wins resultText = `You won! The coin landed on ${coinResult}.`; } else { balance -= betAmount; // User loses resultText = `You lost! The coin landed on ${coinResult}.`; } // Update balance and result text document.getElementById("balance").innerText = balance; document.getElementById("gameResult").innerText = resultText; });
